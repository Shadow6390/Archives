    private void btnRemoverActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnRemoverActionPerformed
        try {
            int i = listMateriais.getSelectedIndex();
            controller.removerMaterial(indexParede, i);
            updateList();
            framePai.revalidate();
        } catch (IllegalArgumentException ex) {
            JOptionPane.showMessageDialog(
                    framePai,
                    ex.getMessage(),
                    Simulacao.getFrasePelaChave("uc1.janela.titulo"),
                    JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_btnRemoverActionPerformed

    private void btnGuardarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnGuardarActionPerformed
        try {
            String opcoes[] = {Simulacao.getFrasePelaChave("menu.sim"), Simulacao.getFrasePelaChave("menu.nao")};
            int resposta = JOptionPane.showOptionDialog(
                    null, Simulacao.getFrasePelaChave("uc1.janela.mensagem"),
                    Simulacao.getFrasePelaChave("uc1.janela.titulo"), 0,
                    JOptionPane.QUESTION_MESSAGE, null, opcoes, opcoes[0]);
            if (resposta == 0 ) {
                double temperatura
                        = Double.parseDouble(this.tempField.getText());
                controller.guardarParede(indexParede, temperatura);
                framePai.revalidate();
                dispose();
            }
            else
            {
                controller.revertParedeChanges(indexParede);
            }
        } catch (IllegalArgumentException ex) {
            JOptionPane.showMessageDialog(
                    framePai,
                    ex.getMessage(),
                    Simulacao.getFrasePelaChave("uc1.janela.titulo"),
                    JOptionPane.ERROR_MESSAGE);
        }

    }//GEN-LAST:event_btnGuardarActionPerformed

    public void updateList()
    {
        modelMat.removeAllElements();
        modelEsp.removeAllElements();
        String[] mats = controller.getListaMateriaisParedes().get(indexParede);
        String[] esps = controller.getListaEspessurasParedes().get(indexParede);
        for (String focus:mats)
        {
            modelMat.addElement(focus);
        }
        for (String focus:esps)
        {
            modelEsp.addElement(focus);
        }
        listMateriais.setModel(modelMat);
        listEspessuras.setModel(modelEsp);
    }
