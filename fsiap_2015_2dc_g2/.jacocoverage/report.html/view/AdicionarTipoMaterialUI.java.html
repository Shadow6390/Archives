    private void btnAdicionarImagemActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnAdicionarImagemActionPerformed

        JFileChooser chooser = new JFileChooser();
        chooser.setAcceptAllFileFilterUsed(false);
        chooser.setFileFilter(new FileNameExtensionFilter("Joint Photographics Group (JPG)","jpg","jpeg"));
        chooser.addChoosableFileFilter(new FileNameExtensionFilter("Portable Network Graphics (PNG)","png"));
        int result = chooser.showOpenDialog(this);
        switch (result)
        {
            case JFileChooser.APPROVE_OPTION:
                txtCaminho.setText(chooser.getSelectedFile().getAbsolutePath());
            break;
        }
    }//GEN-LAST:event_btnAdicionarImagemActionPerformed

    private void btnAdicionarMaterialActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnAdicionarMaterialActionPerformed
        try {
            this.controller.novoMaterial(txtNome.getText(), txtDescricao.getText(),
                    Double.parseDouble(txtCoefCond.getText()),
                    Double.parseDouble(txtCoefConv.getText()),
                    Double.parseDouble(txtCoefRad.getText()), ImageIO.read(new File(txtCaminho.getText())));

            String opcoes[] = {Simulacao.getFrasePelaChave("menu.sim"), Simulacao.getFrasePelaChave("menu.nao")};
            int resposta = JOptionPane.showOptionDialog(
                    null, Simulacao.getFrasePelaChave("uc3.janela.mensagem"),
                    Simulacao.getFrasePelaChave("uc3.janela.titulo"), 0,
                    JOptionPane.QUESTION_MESSAGE, null, opcoes, opcoes[0]);
            if (resposta == JOptionPane.OK_OPTION) {
                if (this.controller.RegistaMaterial())
                {
                    JOptionPane.showMessageDialog(this,Simulacao.getFrasePelaChave("uc3.register.successful"),"",JOptionPane.INFORMATION_MESSAGE);
                    dispose();
                }
                else
                {
                    JOptionPane.showMessageDialog(this,Simulacao.getFrasePelaChave("uc3.register.error"),"",JOptionPane.ERROR_MESSAGE);
                }
            }
        } catch (IllegalArgumentException ex) {
            JOptionPane.showMessageDialog(
                    framePai,
                    ex.getMessage(),
                    Simulacao.getFrasePelaChave("uc3.janela.titulo"),
                    JOptionPane.ERROR_MESSAGE);
        } catch (IOException ex) {
            Logger.getLogger(AdicionarTipoMaterialUI.class.getName()).log(Level.SEVERE, null, ex);
        }
    }//GEN-LAST:event_btnAdicionarMaterialActionPerformed
